  <script>
 const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");
const scoreDisplay = document.getElementById("score");
const recordDisplay = document.getElementById("record");
let snake = [{ x: 200, y: 200 }];
let direction = "RIGHT";
let foods = [spawnFood(), spawnFood()];
let gameInterval;
let score = 0;
let speed = 150;
let record = localStorage.getItem("snakeRecord") || 0;
recordDisplay.textContent = record;
function spawnFood() {
let chance = Math.random();
if (chance < 0.1) return { x: randomPosition(), y: randomPosition(), type: "yellow" }; // 10% chance for yellow
if (chance < 0.15) return { x: randomPosition(), y: randomPosition(), type: "blue" }; // 5% chance for blue
return { x: randomPosition(), y: randomPosition(), type: "red" }; // 85% chance for red
}
function randomPosition() {
return Math.floor(Math.random() * 20) * 20;
}
function drawSnake() {
ctx.fillStyle = "lime";
snake.forEach(segment => {
ctx.fillRect(segment.x, segment.y, 20, 20);
});
}
function drawFood() {
foods.forEach(food => {
ctx.fillStyle = food.type;
ctx.fillRect(food.x, food.y, 20, 20);
});
}
 function moveSnake() {
let head = { ...snake[0] };
if (direction === "UP") head.y -= 20;
if (direction === "DOWN") head.y += 20;
if (direction === "LEFT") head.x -= 20;
if (direction === "RIGHT") head.x += 20;
snake.unshift(head);
for (let i = 0; i < foods.length; i++) {
if (head.x === foods[i].x && head.y === foods[i].y) {
if (foods[i].type === "yellow") {
score += 10;
} else if (foods[i].type === "blue") {
score *= 2;
} else {
score++;
}
scoreDisplay.textContent = score;
if (score > record) {
record = score;
localStorage.setItem("snakeRecord", record);
recordDisplay.textContent = record;
}
snake.push({ x: head.x, y: head.y });
foods[i] = spawnFood();
}
}
if (!foods.some(food => food.x === head.x && food.y === head.y)) {
snake.pop();
}
}
function checkCollision() {
let head = snake[0];
if (head.x < 0 || head.x >= canvas.width || head.y < 0 || head.y >= canvas.height) {
clearInterval(gameInterval);
alert("Game Over! Score: " + score);
location.reload();
}
for (let i = 1; i < snake.length; i++) {
if (head.x === snake[i].x && head.y === snake[i].y) {
clearInterval(gameInterval);
alert("Game Over! Score: " + score);
location.reload();
}
}
}
function updateGame() {
ctx.clearRect(0, 0, canvas.width, canvas.height);
drawFood();
moveSnake();
drawSnake();
checkCollision();
}
window.addEventListener("keydown", (e) => {
if ((e.key === "ArrowUp" || e.key === "w") && direction !== "DOWN") direction = "UP";
if ((e.key === "ArrowDown" || e.key === "s") && direction !== "UP") direction = "DOWN";
if ((e.key === "ArrowLeft" || e.key === "a") && direction !== "RIGHT") direction = "LEFT";
if ((e.key === "ArrowRight" || e.key === "d") && direction !== "LEFT") direction = "RIGHT";
});
gameInterval = setInterval(updateGame, speed);
 </script>

    let score = 0;
    let timeRemaining = 10;
    let gameActive = false;

    const scoreDisplay = document.getElementById('score');
    const startButton = document.getElementById('startButton');

    function startGame() {

      timeRemainingDisplay.textContent = timeRemaining;
      clickButton.disabled = false;
    }

    clickButton.addEventListener('click', () => {
      if (gameActive) { score++; scoreDisplay.textContent = score; }
    });
    startButton.addEventListener('click', startGame);
